    CSS+JS 实现的3D球面标签云
    -

    之前曾经用过一个基于 Flash 实现的 3D球面标签云，中文字体是其中一大问题，毕竟修改 Flash 囊括的中文字体比较麻烦，而涵盖全部中文 Flash 文件又会太大。于是便有了下面这个基于 CSS + JavaScript 实现的 3D球面标签云。

    http://1122sw.cn/images/bqyun.html 演示地址！！

    JavaScript部分代码如下，保存为 3D-Tags.js ：

     

    1. // JavaScript 代码开始  
    2. var radius = 120;  
    3. var dtr = Math.PI/180;  
    4. var d=300;  
    5.    
    6. var mcList = [];  
    7. var active = false;  
    8. var lasta = 1;  
    9. var lastb = 1;  
    10. var distr = true;  
    11. var tspeed=10;  
    12. var size=250;  
    13.    
    14. var mouseX=0;  
    15. var mouseY=0;  
    16.    
    17. var howElliptical=1;  
    18.    
    19. var aA=null;  
    20. var oDiv=null;  
    21.    
    22. window.onload=function ()  
    23. {  
    24.     var i=0;  
    25.     var oTag=null;  
    26.    
    27.     oDiv=document.getElementById('div1');  
    28.    
    29.     aA=oDiv.getElementsByTagName('a');  
    30.    
    31.     for(i=0;i<aA.length;i++)  
    32.     {  
    33.         oTag={};  
    34.    
    35.         oTag.offsetWidth=aA[i].offsetWidth;  
    36.         oTag.offsetHeight=aA[i].offsetHeight;  
    37.    
    38.         mcList.push(oTag);  
    39.     }  
    40.    
    41.     sineCosine( 0,0,0 );  
    42.    
    43.     positionAll();  
    44.    
    45.     oDiv.onmouseover=function ()  
    46.     {  
    47.         active=true;  
    48.     };  
    49.    
    50.     oDiv.onmouseout=function ()  
    51.     {  
    52.         active=false;  
    53.     };  
    54.    
    55.     oDiv.onmousemove=function (ev)  
    56.     {  
    57.         var oEvent=window.event || ev;  
    58.    
    59.         mouseX=oEvent.clientX-(oDiv.offsetLeft+oDiv.offsetWidth/2);  
    60.         mouseY=oEvent.clientY-(oDiv.offsetTop+oDiv.offsetHeight/2);  
    61.    
    62.         mouseX/=5;  
    63.         mouseY/=5;  
    64.     };  
    65.    
    66.     setInterval(update, 30);  
    67. };  
    68.    
    69. function update()  
    70. {  
    71.     var a;  
    72.     var b;  
    73.    
    74.     if(active)  
    75.     {  
    76.         a = (-Math.min( Math.max( -mouseY, -size ), size ) / radius ) * tspeed;  
    77.         b = (Math.min( Math.max( -mouseX, -size ), size ) / radius ) * tspeed;  
    78.     }  
    79.     else  
    80.     {  
    81.         a = lasta * 0.98;  
    82.         b = lastb * 0.98;  
    83.     }  
    84.    
    85.     lasta=a;  
    86.     lastb=b;  
    87.    
    88.     if(Math.abs(a)<=0.01 && Math.abs(b)<=0.01)  
    89.     {  
    90.         return;  
    91.     }  
    92.    
    93.     var c=0;  
    94.     sineCosine(a,b,c);  
    95.     for(var j=0;j<mcList.length;j++)  
    96.     {  
    97.         var rx1=mcList[j].cx;  
    98.         var ry1=mcList[j].cy*ca+mcList[j].cz*(-sa);  
    99.         var rz1=mcList[j].cy*sa+mcList[j].cz*ca;  
    100.    
    101.         var rx2=rx1*cb+rz1*sb;  
    102.         var ry2=ry1;  
    103.         var rz2=rx1*(-sb)+rz1*cb;  
    104.    
    105.         var rx3=rx2*cc+ry2*(-sc);  
    106.         var ry3=rx2*sc+ry2*cc;  
    107.         var rz3=rz2;  
    108.    
    109.         mcList[j].cx=rx3;  
    110.         mcList[j].cy=ry3;  
    111.         mcList[j].cz=rz3;  
    112.    
    113.         per=d/(d+rz3);  
    114.    
    115.         mcList[j].x=(howElliptical*rx3*per)-(howElliptical*2);  
    116.         mcList[j].y=ry3*per;  
    117.         mcList[j].scale=per;  
    118.         mcList[j].alpha=per;  
    119.    
    120.         mcList[j].alpha=(mcList[j].alpha-0.6)*(10/6);  
    121.     }  
    122.    
    123.     doPosition();  
    124.     depthSort();  
    125. }  
    126.    
    127. function depthSort()  
    128. {  
    129.     var i=0;  
    130.     var aTmp=[];  
    131.    
    132.     for(i=0;i<aA.length;i++)  
    133.     {  
    134.         aTmp.push(aA[i]);  
    135.     }  
    136.    
    137.     aTmp.sort  
    138.     (  
    139.         function (vItem1, vItem2)  
    140.         {  
    141.             if(vItem1.cz>vItem2.cz)  
    142.             {  
    143.                 return -1;  
    144.             }  
    145.             else if(vItem1.cz<vItem2.cz)  
    146.             {  
    147.                 return 1;  
    148.             }  
    149.             else  
    150.             {  
    151.                 return 0;  
    152.             }  
    153.         }  
    154.     );  
    155.    
    156.     for(i=0;i<aTmp.length;i++)  
    157.     {  
    158.         aTmp[i].style.zIndex=i;  
    159.     }  
    160. }  
    161.    
    162. function positionAll()  
    163. {  
    164.     var phi=0;  
    165.     var theta=0;  
    166.     var max=mcList.length;  
    167.     var i=0;  
    168.    
    169.     var aTmp=[];  
    170.     var oFragment=document.createDocumentFragment();  
    171.    
    172.     //随机排序  
    173.     for(i=0;i<aA.length;i++)  
    174.     {  
    175.         aTmp.push(aA[i]);  
    176.     }  
    177.    
    178.     aTmp.sort  
    179.     (  
    180.         function ()  
    181.         {  
    182.             return Math.random()<0.5?1:-1;  
    183.         }  
    184.     );  
    185.    
    186.     for(i=0;i<aTmp.length;i++)  
    187.     {  
    188.         oFragment.appendChild(aTmp[i]);  
    189.     }  
    190.    
    191.     oDiv.appendChild(oFragment);  
    192.    
    193.     for( var i=1; i<max+1; i++){  
    194.         if( distr )  
    195.         {  
    196.             phi = Math.acos(-1+(2*i-1)/max);  
    197.             theta = Math.sqrt(max*Math.PI)*phi;  
    198.         }  
    199.         else  
    200.         {  
    201.             phi = Math.random()*(Math.PI);  
    202.             theta = Math.random()*(2*Math.PI);  
    203.         }  
    204.         //坐标变换  
    205.         mcList[i-1].cx = radius * Math.cos(theta)*Math.sin(phi);  
    206.         mcList[i-1].cy = radius * Math.sin(theta)*Math.sin(phi);  
    207.         mcList[i-1].cz = radius * Math.cos(phi);  
    208.    
    209.         aA[i-1].style.left=mcList[i-1].cx+oDiv.offsetWidth/2-mcList[i-1].offsetWidth/2+'px';  
    210.         aA[i-1].style.top=mcList[i-1].cy+oDiv.offsetHeight/2-mcList[i-1].offsetHeight/2+'px';  
    211.     }  
    212. }  
    213.    
    214. function doPosition()  
    215. {  
    216.     var l=oDiv.offsetWidth/2;  
    217.     var t=oDiv.offsetHeight/2;  
    218.     for(var i=0;i<mcList.length;i++)  
    219.     {  
    220.         aA[i].style.left=mcList[i].cx+l-mcList[i].offsetWidth/2+'px';  
    221.         aA[i].style.top=mcList[i].cy+t-mcList[i].offsetHeight/2+'px';  
    222.    
    223.         aA[i].style.fontSize=Math.ceil(12*mcList[i].scale/2)+8+'px';  
    224.    
    225.         aA[i].style.filter="alpha(opacity="+100*mcList[i].alpha+")";  
    226.         aA[i].style.opacity=mcList[i].alpha;  
    227.     }  
    228. }  
    229.    
    230. function sineCosine( a, b, c)  
    231. {  
    232.     sa = Math.sin(a * dtr);  
    233.     ca = Math.cos(a * dtr);  
    234.     sb = Math.sin(b * dtr);  
    235.     cb = Math.cos(b * dtr);  
    236.     sc = Math.sin(c * dtr);  
    237.     cc = Math.cos(c * dtr);  
    238. }
